CREATE TABLE MUSIC_CHART(
NO NUMBER(3) PRIMARY KEY,
TITLE VARCHAR2(100) NOT NULL,
SINGER VARCHAR2(100) NOT NULL,
LYRICIST VARCHAR2(100) NOT NULL,
SONGWRITER VARCHAR2(100) NOT NULL,
RELEASE_DATE DATE,
GENRE VARCHAR2(100),
);
CREATE SEQUENCE MUSIC_SEQ
INCREMENT BY 1
START WITH 1
MINVALUE 1
MAXVALUE 1000
NOCYCLE
NOCACHE;
